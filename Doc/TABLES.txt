CREATE TABLE Produtos (
    ProdutoId int,
    Referencia varchar(255),
    Descricao varchar(255),
    ObservacaoId int
);

CREATE TABLE Lpns (
    LpnId int,
    ProdutoId int,
    Quantidade double(10,3),
    LpnContenedorId int,
    LocalizacaoId int
);

CREATE TABLE Localizacoes (
    LocalizacaoId int,
    LocalizacaoSuperiorId int,
    Divisao varchar(255),
    ObservacaoId int
);

CREATE TABLE Usuarios (
    UsuarioId int,
    Apelido varchar(255),
    Senha varchar(255),
    Supervisor bit
);

CREATE TABLE Observacoes (
    ObservacaoId int,
    Texto text
);
